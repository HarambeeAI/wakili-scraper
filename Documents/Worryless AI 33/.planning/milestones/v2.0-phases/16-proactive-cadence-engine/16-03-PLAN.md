Create `20260320000003_event_detector.sql` with:

1. **check_event_triggers() function** — a SECURITY DEFINER SQL/plpgsql function that:

   a) Checks for **overdue invoices** (invoices WHERE status = 'sent' AND due_date < NOW() AND due_date > NOW() - INTERVAL '7 days'). Maps to agent_type = 'accountant'. Only enqueue if there are 1+ overdue invoices for the user.

   b) Checks for **stale deals** (leads WHERE status NOT IN ('closed_won','closed_lost','rejected') AND updated_at < NOW() - INTERVAL '3 days'). Maps to agent_type = 'sales_rep'. Only enqueue if there are 1+ stale deals.

   c) Checks for **viral posts** (social_posts WHERE published_at > NOW() - INTERVAL '24 hours' AND engagement_count > 0). For viral detection: compare each post's engagement_count against the user's average. If engagement_count > 3 * avg(engagement_count), it's viral. Maps to agent_type = 'marketer'. Note: social_posts may not have engagement_count column — check codebase. If not, skip this detector (will need Phase 17 analytics to work).

   For each detected event:
   - Check cooldown: `user_agents.last_event_notified_at IS NULL OR last_event_notified_at < NOW() - INTERVAL cadence_config->>'event_cooldown_hours' || ' hours'`
   - Check feature flag: join profiles to verify `use_langgraph = true`
   - Check cadence_config: `(cadence_config->>'event_triggers_enabled')::boolean = true`
   - If all pass: enqueue pgmq message with `cadence_tier: 'event'` and `event_type: 'overdue_invoice'|'stale_deal'|'viral_post'`
   - Update `last_event_notified_at = NOW()` on the user_agents row

   Use plpgsql (not pure SQL) because we need IF/LOOP logic for per-user event checks.

   ```sql
   CREATE OR REPLACE FUNCTION public.check_event_triggers()
   RETURNS void
   LANGUAGE plpgsql
   SECURITY DEFINER
   AS $$
   DECLARE
     r RECORD;
     cooldown_interval INTERVAL;
   BEGIN
     -- Event 1: Overdue invoices -> accountant
     FOR r IN
       SELECT DISTINCT ua.id AS user_agent_id, ua.user_id, ua.agent_type_id, ua.cadence_config, ua.last_event_notified_at
       FROM public.user_agents ua
       JOIN public.profiles p ON p.user_id = ua.user_id
       WHERE ua.is_active = true
         AND ua.agent_type_id = 'accountant'
         AND (ua.cadence_config->>'event_triggers_enabled')::boolean = true
         AND p.use_langgraph = true
         AND EXISTS (
           SELECT 1 FROM public.invoices i
           WHERE i.user_id = ua.user_id
             AND i.status = 'sent'
             AND i.due_date < NOW()
             AND i.due_date > NOW() - INTERVAL '7 days'
         )
     LOOP
       cooldown_interval := ((r.cadence_config->>'event_cooldown_hours')::int || ' hours')::interval;
       IF r.last_event_notified_at IS NULL OR r.last_event_notified_at < NOW() - cooldown_interval THEN
         PERFORM pgmq_public.send(
           'heartbeat_jobs',
           jsonb_build_object(
             'user_agent_id', r.user_agent_id,
             'user_id', r.user_id,
             'agent_type_id', r.agent_type_id,
             'cadence_tier', 'event',
             'event_type', 'overdue_invoice'
           ),
           0
         );
         UPDATE public.user_agents SET last_event_notified_at = NOW() WHERE id = r.user_agent_id;
       END IF;
     END LOOP;

     -- Event 2: Stale deals -> sales_rep
     FOR r IN
       SELECT DISTINCT ua.id AS user_agent_id, ua.user_id, ua.agent_type_id, ua.cadence_config, ua.last_event_notified_at
       FROM public.user_agents ua
       JOIN public.profiles p ON p.user_id = ua.user_id
       WHERE ua.is_active = true
         AND ua.agent_type_id = 'sales_rep'
         AND (ua.cadence_config->>'event_triggers_enabled')::boolean = true
         AND p.use_langgraph = true
         AND EXISTS (
           SELECT 1 FROM public.leads l
           WHERE l.user_id = ua.user_id
             AND l.status NOT IN ('closed_won','closed_lost','rejected')
             AND l.updated_at < NOW() - INTERVAL '3 days'
         )
     LOOP
       cooldown_interval := ((r.cadence_config->>'event_cooldown_hours')::int || ' hours')::interval;
       IF r.last_event_notified_at IS NULL OR r.last_event_notified_at < NOW() - cooldown_interval THEN
         PERFORM pgmq_public.send(
           'heartbeat_jobs',
           jsonb_build_object(
             'user_agent_id', r.user_agent_id,
             'user_id', r.user_id,
             'agent_type_id', r.agent_type_id,
             'cadence_tier', 'event',
             'event_type', 'stale_deal'
           ),
           0
         );
         UPDATE public.user_agents SET last_event_notified_at = NOW() WHERE id = r.user_agent_id;
       END IF;
     END LOOP;

     -- Event 3: Overdue contracts -> legal_advisor
     FOR r IN
       SELECT DISTINCT ua.id AS user_agent_id, ua.user_id, ua.agent_type_id, ua.cadence_config, ua.last_event_notified_at
       FROM public.user_agents ua
       JOIN public.profiles p ON p.user_id = ua.user_id
       WHERE ua.is_active = true
         AND ua.agent_type_id = 'legal_advisor'
         AND (ua.cadence_config->>'event_triggers_enabled')::boolean = true
         AND p.use_langgraph = true
         AND EXISTS (
           SELECT 1 FROM public.contracts c
           WHERE c.user_id = ua.user_id
             AND c.expiry_date < NOW() + INTERVAL '7 days'
             AND c.expiry_date > NOW()
             AND c.status = 'active'
         )
     LOOP
       cooldown_interval := ((r.cadence_config->>'event_cooldown_hours')::int || ' hours')::interval;
       IF r.last_event_notified_at IS NULL OR r.last_event_notified_at < NOW() - cooldown_interval THEN
         PERFORM pgmq_public.send(
           'heartbeat_jobs',
           jsonb_build_object(
             'user_agent_id', r.user_agent_id,
             'user_id', r.user_id,
             'agent_type_id', r.agent_type_id,
             'cadence_tier', 'event',
             'event_type', 'expiring_contract'
           ),
           0
         );
         UPDATE public.user_agents SET last_event_notified_at = NOW() WHERE id = r.user_agent_id;
       END IF;
     END LOOP;
   END;
   $$;

   GRANT EXECUTE ON FUNCTION public.check_event_triggers() TO service_role;
   ```

```typescript
export const EVENT_TYPES = {
  OVERDUE_INVOICE: 'overdue_invoice',
  STALE_DEAL: 'stale_deal',
  EXPIRING_CONTRACT: 'expiring_contract',
} as const;

export type EventType = typeof EVENT_TYPES[keyof typeof EVENT_TYPES];

export interface EventPromptConfig {
  agentType: string;
  prompt: string;
}

export const EVENT_PROMPTS: Record<EventType, EventPromptConfig> = {
  overdue_invoice: {
    agentType: 'accountant',
    prompt: 'URGENT EVENT: Invoices past due date detected. List all invoices with outstanding balances that are past their due date. Check cashflow impact and flag the severity. Do NOT send reminders — only surface findings for my review.',
  },
  stale_deal: {
    agentType: 'sales_rep',
    prompt: 'URGENT EVENT: Stale deals detected — leads with no activity in 3+ days. Detect stale deals in the pipeline, check email engagement on recent outreach for those leads, and provide a re-engagement recommendation. Do NOT send any outreach.',
  },
  expiring_contract: {
    agentType: 'legal_advisor',
    prompt: 'URGENT EVENT: Contracts expiring within 7 days detected. Review the contract calendar for upcoming expirations. Flag any contracts that need renewal action. Surface findings for review.',
  },
};

export function getEventPrompt(eventType: string): EventPromptConfig | null {
  return EVENT_PROMPTS[eventType as EventType] ?? null;
}
```
