    ```typescript
    import { createClient } from "https://esm.sh/@supabase/supabase-js@2.86.0";

    // TODO(Phase 5): Per-user timezone scheduling — currently fires at 8am UTC
    // Phase 5 blocker: per-user next-run timestamp approach deferred per STATE.md

    const SUPABASE_URL = Deno.env.get("SUPABASE_URL") ?? "";
    const SUPABASE_SERVICE_ROLE_KEY = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY") ?? "";

    Deno.serve(async (_req) => {
      const supabaseAdmin = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY);
      const since = new Date(Date.now() - 24 * 60 * 60 * 1000).toISOString();

      // Fetch all digest-severity heartbeat log rows from past 24 hours
      const { data: digestRows, error } = await supabaseAdmin
        .from('agent_heartbeat_log')
        .select('user_id, agent_type_id, summary, run_at')
        .eq('severity', 'digest')
        .gte('run_at', since)
        .order('user_id', { ascending: true });

      if (error) {
        console.error('Digest query failed:', error);
        return new Response(JSON.stringify({ error: error.message }), { status: 500 });
      }

      if (!digestRows || digestRows.length === 0) {
        console.log('No digest findings in past 24h — nothing to send');
        return new Response(JSON.stringify({ sent: 0 }), { status: 200 });
      }

      // Group by user_id
      const byUser = new Map<string, typeof digestRows>();
      for (const row of digestRows) {
        const existing = byUser.get(row.user_id) ?? [];
        existing.push(row);
        byUser.set(row.user_id, existing);
      }

      let sent = 0;
      for (const [userId, rows] of byUser.entries()) {
        // Build markdown briefing
        const bulletPoints = rows
          .map(r => `- **${r.agent_type_id}**: ${r.summary}`)
          .join('\n');
        const briefingContent = `## Morning Briefing\n\nHere are the items your agents flagged for your review:\n\n${bulletPoints}\n\nAll of these were low-priority findings from the past 24 hours. Let me know if you'd like to act on any of them.`;

        // Insert as Chief of Staff message in existing messages table
        // Use the same insert pattern as send-daily-briefing
        await supabaseAdmin.from('messages').insert({
          user_id: userId,
          role: 'assistant',
          content: briefingContent,
          agent_type: 'chief_of_staff',
          created_at: new Date().toISOString(),
        });

        sent++;
      }

      console.log(`Morning digest: sent briefings to ${sent} users`);
      return new Response(JSON.stringify({ sent }), { status: 200 });
    });
    ```
