    ```typescript
    import { createClient } from "https://esm.sh/@supabase/supabase-js@2.86.0";

    const SUPABASE_URL = Deno.env.get("SUPABASE_URL") ?? "";
    const SUPABASE_SERVICE_ROLE_KEY = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY") ?? "";

    Deno.serve(async (_req) => {
      // SEC-02: Request body is entirely ignored.
      // User identity is derived exclusively from user_agents DB rows.
      const supabaseAdmin = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY);

      // Find agents due for a heartbeat within active hours and under daily budget.
      // Business-hours check: AT TIME ZONE with IANA string from profiles.timezone.
      // Budget check: COUNT aggregate on agent_heartbeat_log (self-resetting, no counter column).
      const { data: dueAgents, error } = await supabaseAdmin.rpc('get_due_heartbeat_agents');

      if (error) {
        console.error('Dispatcher query failed:', error);
        return new Response(JSON.stringify({ error: error.message }), { status: 500 });
      }

      let enqueued = 0;
      const skippedBudget: string[] = [];

      for (const agent of dueAgents ?? []) {
        // Enqueue into pgmq heartbeat_jobs queue
        const { error: sendError } = await supabaseAdmin
          .schema('pgmq_public')
          .rpc('send', {
            queue_name: 'heartbeat_jobs',
            message: {
              userAgentId: agent.id,
              userId: agent.user_id,        // from DB row — not from request
              agentTypeId: agent.agent_type_id,
            },
            sleep_seconds: 0,
          });

        if (sendError) {
          console.error(`Failed to enqueue agent ${agent.id}:`, sendError);
          continue;
        }

        // Advance next_heartbeat_at ONLY for successfully enqueued agents.
        // Budget-skipped agents retain existing next_heartbeat_at (Pitfall 6 prevention).
        await supabaseAdmin
          .from('user_agents')
          .update({
            next_heartbeat_at: new Date(
              Date.now() + agent.heartbeat_interval_hours * 60 * 60 * 1000
            ).toISOString(),
          })
          .eq('id', agent.id);

        enqueued++;
      }

      console.log(`Dispatcher: enqueued=${enqueued}`);
      return new Response(JSON.stringify({ enqueued }), {
        status: 200,
        headers: { 'Content-Type': 'application/json' },
      });
    });
    ```

    IMPORTANT: The query above uses `supabaseAdmin.rpc('get_due_heartbeat_agents')` — this requires a Postgres function to avoid complex AT TIME ZONE logic in TypeScript. Add the function definition to migration 20260313000006 (or a new migration 20260313000008):

    ```sql
    -- Add to the end of migration 20260313000006 (or new migration 20260313000008):
    CREATE OR REPLACE FUNCTION public.get_due_heartbeat_agents()
    RETURNS TABLE (
      id UUID,
      user_id UUID,
      agent_type_id TEXT,
      heartbeat_interval_hours INTEGER
    )
    LANGUAGE sql
    SECURITY DEFINER
    AS $$
      SELECT
        ua.id,
        ua.user_id,
        ua.agent_type_id,
        ua.heartbeat_interval_hours
      FROM public.user_agents ua
      JOIN public.profiles p ON p.user_id = ua.user_id
      WHERE
        ua.is_active = true
        AND ua.heartbeat_enabled = true
        AND ua.next_heartbeat_at <= now()
        AND EXTRACT(HOUR FROM (now() AT TIME ZONE COALESCE(p.timezone, 'UTC')))
            >= EXTRACT(HOUR FROM ua.heartbeat_active_hours_start)
        AND EXTRACT(HOUR FROM (now() AT TIME ZONE COALESCE(p.timezone, 'UTC')))
            < EXTRACT(HOUR FROM ua.heartbeat_active_hours_end)
        AND (
          SELECT COUNT(*) FROM public.agent_heartbeat_log ahl
          WHERE ahl.user_id = ua.user_id
            AND ahl.agent_type_id = ua.agent_type_id
            AND ahl.run_at >= (
              date_trunc('day', now() AT TIME ZONE COALESCE(p.timezone, 'UTC'))
              AT TIME ZONE COALESCE(p.timezone, 'UTC')
            )
        ) < COALESCE(ua.heartbeat_daily_budget, 6)
      ORDER BY ua.next_heartbeat_at ASC
      LIMIT 50;
    $$;
    ```
