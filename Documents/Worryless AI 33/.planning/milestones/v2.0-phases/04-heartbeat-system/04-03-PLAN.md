    ```typescript
    import { useState, useEffect, useCallback } from 'react';
    import { supabase } from '@/integrations/supabase/client';

    export interface HeartbeatConfig {
      heartbeat_enabled: boolean;
      heartbeat_interval_hours: 1 | 2 | 4 | 8;
      heartbeat_active_hours_start: string; // "HH:MM"
      heartbeat_active_hours_end: string;   // "HH:MM"
    }

    export function useHeartbeatConfig(agentTypeId: string) {
      const [config, setConfig] = useState<HeartbeatConfig | null>(null);
      const [isLoading, setIsLoading] = useState(true);
      const [isSaving, setIsSaving] = useState(false);

      useEffect(() => {
        let cancelled = false;
        async function load() {
          setIsLoading(true);
          const { data: { user } } = await supabase.auth.getUser();
          if (!user) { setIsLoading(false); return; }

          const { data } = await (supabase as any)
            .from('user_agents')
            .select('heartbeat_enabled, heartbeat_interval_hours, heartbeat_active_hours_start, heartbeat_active_hours_end')
            .eq('user_id', user.id)
            .eq('agent_type_id', agentTypeId)
            .single();

          if (!cancelled && data) {
            setConfig({
              heartbeat_enabled: data.heartbeat_enabled,
              heartbeat_interval_hours: data.heartbeat_interval_hours as 1 | 2 | 4 | 8,
              heartbeat_active_hours_start: data.heartbeat_active_hours_start ?? '08:00',
              heartbeat_active_hours_end: data.heartbeat_active_hours_end ?? '20:00',
            });
          }
          if (!cancelled) setIsLoading(false);
        }
        load();
        return () => { cancelled = true; };
      }, [agentTypeId]);

      const updateConfig = useCallback(async (patch: Partial<HeartbeatConfig>) => {
        const { data: { user } } = await supabase.auth.getUser();
        if (!user) return;
        setIsSaving(true);
        await (supabase as any)
          .from('user_agents')
          .update(patch)
          .eq('user_id', user.id)
          .eq('agent_type_id', agentTypeId);
        setConfig(prev => prev ? { ...prev, ...patch } : prev);
        setIsSaving(false);
      }, [agentTypeId]);

      return { config, isLoading, isSaving, updateConfig };
    }
    ```
