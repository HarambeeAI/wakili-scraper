    Add VAPID push send INSIDE the same `if (severity === "urgent")` block, AFTER the sendUrgentEmail call. Use a try/catch so push failure never breaks email delivery or the heartbeat run:

    ```typescript
    // Send VAPID push notification
    try {
      const vapidPublicKey = Deno.env.get("VAPID_PUBLIC_KEY");
      const vapidPrivateKey = Deno.env.get("VAPID_PRIVATE_KEY");
      if (vapidPublicKey && vapidPrivateKey) {
        const webpush = await import("jsr:@negrel/webpush");
        const appServer = await webpush.ApplicationServer.new({
          contactInformation: "mailto:noreply@worryless.ai",
          vapidKeys: { publicKey: vapidPublicKey, privateKey: vapidPrivateKey },
        });
        const { data: subs } = await supabaseAdmin
          .from("push_subscriptions")
          .select("endpoint, p256dh, auth")
          .eq("user_id", userId);
        for (const sub of subs ?? []) {
          const subscriber = appServer.subscribe({
            endpoint: sub.endpoint,
            keys: { p256dh: sub.p256dh, auth: sub.auth },
          });
          await subscriber.pushTextMessage(
            JSON.stringify({ title: agentDisplayName, body: finding }),
            {}
          );
        }
      }
    } catch (pushErr) {
      console.error("[heartbeat-runner] VAPID push failed (non-fatal):", pushErr);
    }
    ```

    Note: VAPID_PUBLIC_KEY and VAPID_PRIVATE_KEY are optional at runtime — if not set, push is silently skipped. This allows the function to deploy before VAPID keys are configured. The user_setup frontmatter declares what the user needs to set.

    No other changes to heartbeat-runner. Existing email, log, and notification insert logic unchanged.
