       ```
       GEMINI_API_KEY=<user-provided-value>
       FIRECRAWL_API_KEY=<user-provided-value>
       APIFY_API_TOKEN=<user-provided-value>
       RESEND_API_KEY=<user-provided-value>

       DATABASE_URL=${{Postgres.DATABASE_URL}}

       ```
       VITE_LOGTO_ENDPOINT=https://${{Logto.RAILWAY_PUBLIC_DOMAIN}}
