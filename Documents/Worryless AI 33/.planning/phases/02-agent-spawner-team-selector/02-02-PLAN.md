```typescript
export function extractJson(raw: string): string {
  return raw.replace(/^```(?:json)?\n?/, '').replace(/\n?```$/, '').trim();
}

export function filterRecommendations(
  parsed: unknown,
  validIds: Set<string>,
  defaultIds: Set<string>
): Array<{agent_type_id: string; reasoning: string; first_week_value: string}> {
  if (!parsed || typeof parsed !== 'object' || !Array.isArray((parsed as any).recommendations)) {
    return [];
  }
  return ((parsed as any).recommendations as any[])
    .filter(r => r?.agent_type_id && validIds.has(r.agent_type_id) && !defaultIds.has(r.agent_type_id))
    .map(r => ({
      agent_type_id: String(r.agent_type_id),
      reasoning: String(r.reasoning || ''),
      first_week_value: String(r.first_week_value || ''),
    }))
    .slice(0, 5);
}
```
