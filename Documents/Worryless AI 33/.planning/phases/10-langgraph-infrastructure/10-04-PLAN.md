```typescript
import { useEffect, useState } from "react";
import { supabase } from "@/integrations/supabase/client";

/**
 * Hook to read the use_langgraph feature flag from the user's profile.
 *
 * Returns:
 * - useLangGraph: boolean (false by default until loaded and flag is true)
 * - loading: boolean
 * - error: string | null
 *
 * Usage in ChatInterface or routing logic:
 *   const { useLangGraph } = useLangGraphFlag();
 *   const endpoint = useLangGraph
 *     ? `${SUPABASE_URL}/functions/v1/langgraph-proxy/invoke`
 *     : `${SUPABASE_URL}/functions/v1/orchestrator`;
 */
export function useLangGraphFlag() {
  const [useLangGraph, setUseLangGraph] = useState<boolean>(false);
  const [loading, setLoading] = useState<boolean>(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let cancelled = false;

    async function fetchFlag() {
      try {
        const { data: { user } } = await supabase.auth.getUser();
        if (!user || cancelled) {
          setLoading(false);
          return;
        }

        const { data, error: queryError } = await supabase
          .from("profiles")
          .select("use_langgraph")
          .eq("user_id", user.id)
          .single();

        if (cancelled) return;

        if (queryError) {
          console.error("[useLangGraphFlag] Query error:", queryError.message);
          setError(queryError.message);
          // Default to false (legacy) on error
          setUseLangGraph(false);
        } else {
          setUseLangGraph(data?.use_langgraph ?? false);
        }
      } catch (err) {
        if (!cancelled) {
          console.error("[useLangGraphFlag] Unexpected error:", err);
          setError(err instanceof Error ? err.message : "Unknown error");
          setUseLangGraph(false);
        }
      } finally {
        if (!cancelled) setLoading(false);
      }
    }

    fetchFlag();

    return () => {
      cancelled = true;
    };
  }, []);

  return { useLangGraph, loading, error };
}

/**
 * Helper to get the correct chat endpoint based on the feature flag.
 * Returns the full URL for either the legacy orchestrator or the langgraph-proxy.
 */
export function getChatEndpoint(useLangGraph: boolean): string {
  const supabaseUrl = import.meta.env.VITE_SUPABASE_URL;
  return useLangGraph
    ? `${supabaseUrl}/functions/v1/langgraph-proxy/invoke`
    : `${supabaseUrl}/functions/v1/orchestrator`;
}
```

This hook:
1. Fetches the user's profile on mount
2. Reads the `use_langgraph` boolean column
3. Defaults to `false` on any error (safe fallback = legacy behavior)
4. Provides a `getChatEndpoint` helper that resolves the correct Edge Function URL
5. Uses the `cancelled` pattern for cleanup (consistent with other hooks in the codebase)
6. Does NOT modify ChatInterface.tsx or any other existing file -- the hook is purely additive
7. The actual wiring into ChatInterface.tsx will happen in Phase 17 when the new chat UI is built

The hook follows the exact same patterns as `useHeartbeatConfig` and `useNotifications`: useState for state, useEffect with async function, supabase query, cleanup on unmount.
