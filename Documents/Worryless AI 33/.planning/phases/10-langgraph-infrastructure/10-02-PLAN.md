let checkpointerInstance: PostgresSaver | null = null;

export async function getCheckpointer(): Promise<PostgresSaver> {
  if (checkpointerInstance) return checkpointerInstance;

  const connectionString = process.env.DATABASE_URL;
  if (!connectionString) {
    throw new Error("DATABASE_URL environment variable is required");
  }

  checkpointerInstance = PostgresSaver.fromConnString(connectionString);
  // Setup creates the checkpoint tables if they don't exist
  await checkpointerInstance.setup();

  console.log("[checkpointer] PostgresSaver initialized and connected");
  return checkpointerInstance;
}

**2. `worrylesssuperagent/langgraph-server/src/persistence/store.ts`:**

```typescript
import pg from "pg";

const { Pool } = pg;

export interface StoreItem {
  prefix: string;
  key: string;
  value: Record<string, unknown>;
  created_at: string;
  updated_at: string;
}

let pool: pg.Pool | null = null;

function getPool(): pg.Pool {
  if (pool) return pool;
  const connectionString = process.env.DATABASE_URL;
  if (!connectionString) {
    throw new Error("DATABASE_URL environment variable is required");
  }
  pool = new Pool({ connectionString, max: 10 });
  return pool;
}

/**
 * Write a value to the LangGraph Store.
 * Namespace format: (user_id, "agent_memory", agent_type)
 * Key: the memory key within that namespace
 */
export async function putStore(
  prefix: string,
  key: string,
  value: Record<string, unknown>
): Promise<void> {
  const db = getPool();
  await db.query(
    `INSERT INTO langgraph.store (prefix, key, value, updated_at)
     VALUES ($1, $2, $3, NOW())
     ON CONFLICT (prefix, key)
     DO UPDATE SET value = $3, updated_at = NOW()`,
    [prefix, key, JSON.stringify(value)]
  );
}

/**
 * Get a value from the LangGraph Store by exact prefix + key.
 */
export async function getStore(
  prefix: string,
  key: string
): Promise<StoreItem | null> {
  const db = getPool();
  const result = await db.query(
    `SELECT prefix, key, value, created_at, updated_at
     FROM langgraph.store
     WHERE prefix = $1 AND key = $2`,
    [prefix, key]
  );
  if (result.rows.length === 0) return null;
  return result.rows[0] as StoreItem;
}

/**
 * Search the Store by prefix (list all keys under a namespace).
 */
export async function searchStore(prefix: string): Promise<StoreItem[]> {
  const db = getPool();
  const result = await db.query(
    `SELECT prefix, key, value, created_at, updated_at
     FROM langgraph.store
     WHERE prefix = $1
     ORDER BY updated_at DESC`,
    [prefix]
  );
  return result.rows as StoreItem[];
}

/**
 * Delete a value from the Store.
 */
export async function deleteStore(prefix: string, key: string): Promise<void> {
  const db = getPool();
  await db.query(
    `DELETE FROM langgraph.store WHERE prefix = $1 AND key = $2`,
    [prefix, key]
  );
}

/**
 * Health check: verify Store table is accessible.
 */
export async function storeHealthCheck(): Promise<boolean> {
  try {
    const db = getPool();
    await db.query("SELECT 1 FROM langgraph.store LIMIT 0");
    return true;
  } catch {
    return false;
  }
}
```

**3. `worrylesssuperagent/langgraph-server/src/graph/echo.ts`:**

```typescript
import { StateGraph, Annotation, MessagesAnnotation } from "@langchain/langgraph";
import type { PostgresSaver } from "@langchain/langgraph-checkpoint-postgres";

/**
 * Minimal echo graph for infrastructure validation.
 * Accepts a message, echoes it back with a prefix.
 * Used to prove: server runs, graph compiles, checkpointer persists state.
 *
 * This graph will be replaced by the real agent topology in Phase 11.
 */

const EchoState = Annotation.Root({
  ...MessagesAnnotation.spec,
});

function echoNode(state: typeof EchoState.State) {
  const lastMessage = state.messages[state.messages.length - 1];
  const content = typeof lastMessage.content === "string"
    ? lastMessage.content
    : JSON.stringify(lastMessage.content);

  return {
    messages: [
      {
        role: "assistant" as const,
        content: `[echo] ${content}`,
      },
    ],
  };
}

export function createEchoGraph(checkpointer: PostgresSaver) {
  const graph = new StateGraph(EchoState)
    .addNode("echo", echoNode)
    .addEdge("__start__", "echo")
    .addEdge("echo", "__end__");

  return graph.compile({ checkpointer });
}
```

**4. `worrylesssuperagent/langgraph-server/src/index.ts`:**

```typescript
import express from "express";
import { getCheckpointer } from "./persistence/checkpointer.js";
import { storeHealthCheck, putStore, getStore, searchStore } from "./persistence/store.js";
import { createEchoGraph } from "./graph/echo.js";
import { HumanMessage } from "@langchain/core/messages";

const app = express();
app.use(express.json());

const PORT = parseInt(process.env.PORT || "3001", 10);

// ── Health Check ──────────────────────────────────────────────────────
app.get("/health", async (_req, res) => {
  try {
    const checkpointer = await getCheckpointer();
    const storeOk = await storeHealthCheck();
    res.json({
      status: "ok",
      version: "1.0.0",
      checkpointer: checkpointer ? "connected" : "disconnected",
      store: storeOk ? "connected" : "disconnected",
      timestamp: new Date().toISOString(),
    });
  } catch (error) {
    res.status(503).json({
      status: "error",
      error: error instanceof Error ? error.message : "Unknown error",
      timestamp: new Date().toISOString(),
    });
  }
});

// ── Echo Graph Invoke (infrastructure validation) ─────────────────────
app.post("/invoke", async (req, res) => {
  try {
    const { message, thread_id } = req.body;
    if (!message || typeof message !== "string") {
      res.status(400).json({ error: "message (string) is required" });
      return;
    }

    const checkpointer = await getCheckpointer();
    const graph = createEchoGraph(checkpointer);

    const config = {
      configurable: {
        thread_id: thread_id || `echo-${Date.now()}`,
      },
    };

    const result = await graph.invoke(
      { messages: [new HumanMessage(message)] },
      config
    );

    const lastMsg = result.messages[result.messages.length - 1];

    res.json({
      response: typeof lastMsg.content === "string" ? lastMsg.content : JSON.stringify(lastMsg.content),
      thread_id: config.configurable.thread_id,
      message_count: result.messages.length,
    });
  } catch (error) {
    console.error("[invoke] Error:", error);
    res.status(500).json({
      error: error instanceof Error ? error.message : "Unknown error",
    });
  }
});

// ── Store endpoints (infrastructure validation) ───────────────────────
app.post("/store", async (req, res) => {
  try {
    const { prefix, key, value } = req.body;
    if (!prefix || !key || !value) {
      res.status(400).json({ error: "prefix, key, and value are required" });
      return;
    }
    await putStore(prefix, key, value);
    res.json({ status: "ok", prefix, key });
  } catch (error) {
    res.status(500).json({ error: error instanceof Error ? error.message : "Unknown error" });
  }
});

app.get("/store/:prefix", async (req, res) => {
  try {
    const items = await searchStore(req.params.prefix);
    res.json({ items });
  } catch (error) {
    res.status(500).json({ error: error instanceof Error ? error.message : "Unknown error" });
  }
});

app.get("/store/:prefix/:key", async (req, res) => {
  try {
    const item = await getStore(req.params.prefix, req.params.key);
    if (!item) {
      res.status(404).json({ error: "Not found" });
      return;
    }
    res.json(item);
  } catch (error) {
    res.status(500).json({ error: error instanceof Error ? error.message : "Unknown error" });
  }
});

// ── Start Server ──────────────────────────────────────────────────────
app.listen(PORT, () => {
  console.log(`[langgraph-server] Running on port ${PORT}`);
  console.log(`[langgraph-server] Health: http://localhost:${PORT}/health`);
});
```
