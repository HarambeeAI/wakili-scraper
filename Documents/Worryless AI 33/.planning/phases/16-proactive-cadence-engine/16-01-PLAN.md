```sql
-- Add cadence_config JSONB with default tiers
ALTER TABLE public.user_agents
  ADD COLUMN IF NOT EXISTS cadence_config JSONB NOT NULL DEFAULT '{
    "daily_enabled": true,
    "weekly_enabled": true,
    "monthly_enabled": false,
    "quarterly_enabled": false,
    "event_triggers_enabled": true,
    "event_cooldown_hours": 4
  }'::jsonb;

-- Add next-run timestamp columns for weekly/monthly/quarterly tiers
ALTER TABLE public.user_agents
  ADD COLUMN IF NOT EXISTS next_weekly_heartbeat_at TIMESTAMPTZ,
  ADD COLUMN IF NOT EXISTS next_monthly_heartbeat_at TIMESTAMPTZ,
  ADD COLUMN IF NOT EXISTS next_quarterly_heartbeat_at TIMESTAMPTZ;

-- Add last_event_notified_at for event-trigger cooldown tracking
ALTER TABLE public.user_agents
  ADD COLUMN IF NOT EXISTS last_event_notified_at TIMESTAMPTZ;

-- Backfill: set initial next-run times for existing active agents
-- Weekly: next Monday 8am in user's timezone
-- Monthly: 1st of next month 8am
-- Quarterly: 1st of next quarter 8am
UPDATE public.user_agents ua
SET
  next_weekly_heartbeat_at = CASE
    WHEN (ua.cadence_config->>'weekly_enabled')::boolean = true
    THEN date_trunc('week', now()) + INTERVAL '7 days' + INTERVAL '8 hours'
    ELSE NULL
  END,
  next_monthly_heartbeat_at = CASE
    WHEN (ua.cadence_config->>'monthly_enabled')::boolean = true
    THEN date_trunc('month', now()) + INTERVAL '1 month' + INTERVAL '8 hours'
    ELSE NULL
  END,
  next_quarterly_heartbeat_at = CASE
    WHEN (ua.cadence_config->>'quarterly_enabled')::boolean = true
    THEN date_trunc('quarter', now()) + INTERVAL '3 months' + INTERVAL '8 hours'
    ELSE NULL
  END
WHERE ua.is_active = true;
```

**Migration 2: 20260320000002_cadence_dispatcher_v2.sql**

Create a new function `get_due_cadence_agents()` that returns agents due for ANY cadence tier (daily, weekly, monthly, quarterly). It must return a `cadence_tier TEXT` column in its output so the runner knows which prompt to use. The function checks:
- For daily: same logic as existing `get_due_heartbeat_agents()` (next_heartbeat_at <= now() + business hours + daily budget)
- For weekly: next_weekly_heartbeat_at <= now() + business hours + cadence_config->>'weekly_enabled' = 'true'
- For monthly: next_monthly_heartbeat_at <= now() + cadence_config->>'monthly_enabled' = 'true'
- For quarterly: next_quarterly_heartbeat_at <= now() + cadence_config->>'quarterly_enabled' = 'true'

Use UNION ALL across the 4 tiers. Each row includes the cadence_tier string. Business-hours check applies to all tiers. Daily budget check applies ONLY to daily tier (weekly/monthly/quarterly are inherently budget-limited by their frequency).

```sql
CREATE OR REPLACE FUNCTION public.get_due_cadence_agents()
RETURNS TABLE (
  id UUID,
  user_id UUID,
  agent_type_id TEXT,
  heartbeat_interval_hours INTEGER,
  cadence_tier TEXT
)
LANGUAGE sql
SECURITY DEFINER
AS $$
  -- Daily tier (same as original get_due_heartbeat_agents)
  SELECT ua.id, ua.user_id, ua.agent_type_id, ua.heartbeat_interval_hours, 'daily'::text AS cadence_tier
  FROM public.user_agents ua
  JOIN public.profiles p ON p.user_id = ua.user_id
  WHERE ua.is_active = true AND ua.heartbeat_enabled = true
    AND (ua.cadence_config->>'daily_enabled')::boolean = true
    AND ua.next_heartbeat_at <= now()
    AND EXTRACT(HOUR FROM (now() AT TIME ZONE COALESCE(p.timezone, 'UTC')))
        >= EXTRACT(HOUR FROM ua.heartbeat_active_hours_start)
    AND EXTRACT(HOUR FROM (now() AT TIME ZONE COALESCE(p.timezone, 'UTC')))
        < EXTRACT(HOUR FROM ua.heartbeat_active_hours_end)
    AND (SELECT COUNT(*) FROM public.agent_heartbeat_log ahl
         WHERE ahl.user_id = ua.user_id AND ahl.agent_type_id = ua.agent_type_id
           AND ahl.run_at >= (date_trunc('day', now() AT TIME ZONE COALESCE(p.timezone, 'UTC'))
                              AT TIME ZONE COALESCE(p.timezone, 'UTC')))
        < COALESCE(ua.heartbeat_daily_budget, 6)

  UNION ALL

  -- Weekly tier
  SELECT ua.id, ua.user_id, ua.agent_type_id, ua.heartbeat_interval_hours, 'weekly'::text
  FROM public.user_agents ua
  JOIN public.profiles p ON p.user_id = ua.user_id
  WHERE ua.is_active = true AND ua.heartbeat_enabled = true
    AND (ua.cadence_config->>'weekly_enabled')::boolean = true
    AND ua.next_weekly_heartbeat_at IS NOT NULL
    AND ua.next_weekly_heartbeat_at <= now()
    AND EXTRACT(HOUR FROM (now() AT TIME ZONE COALESCE(p.timezone, 'UTC')))
        >= EXTRACT(HOUR FROM ua.heartbeat_active_hours_start)
    AND EXTRACT(HOUR FROM (now() AT TIME ZONE COALESCE(p.timezone, 'UTC')))
        < EXTRACT(HOUR FROM ua.heartbeat_active_hours_end)

  UNION ALL

  -- Monthly tier
  SELECT ua.id, ua.user_id, ua.agent_type_id, ua.heartbeat_interval_hours, 'monthly'::text
  FROM public.user_agents ua
  JOIN public.profiles p ON p.user_id = ua.user_id
  WHERE ua.is_active = true AND ua.heartbeat_enabled = true
    AND (ua.cadence_config->>'monthly_enabled')::boolean = true
    AND ua.next_monthly_heartbeat_at IS NOT NULL
    AND ua.next_monthly_heartbeat_at <= now()
    AND EXTRACT(HOUR FROM (now() AT TIME ZONE COALESCE(p.timezone, 'UTC')))
        >= EXTRACT(HOUR FROM ua.heartbeat_active_hours_start)
    AND EXTRACT(HOUR FROM (now() AT TIME ZONE COALESCE(p.timezone, 'UTC')))
        < EXTRACT(HOUR FROM ua.heartbeat_active_hours_end)

  UNION ALL

  -- Quarterly tier
  SELECT ua.id, ua.user_id, ua.agent_type_id, ua.heartbeat_interval_hours, 'quarterly'::text
  FROM public.user_agents ua
  JOIN public.profiles p ON p.user_id = ua.user_id
  WHERE ua.is_active = true AND ua.heartbeat_enabled = true
    AND (ua.cadence_config->>'quarterly_enabled')::boolean = true
    AND ua.next_quarterly_heartbeat_at IS NOT NULL
    AND ua.next_quarterly_heartbeat_at <= now()
    AND EXTRACT(HOUR FROM (now() AT TIME ZONE COALESCE(p.timezone, 'UTC')))
        >= EXTRACT(HOUR FROM ua.heartbeat_active_hours_start)
    AND EXTRACT(HOUR FROM (now() AT TIME ZONE COALESCE(p.timezone, 'UTC')))
        < EXTRACT(HOUR FROM ua.heartbeat_active_hours_end)

  ORDER BY cadence_tier, 5  -- daily first, then weekly, monthly, quarterly
  LIMIT 50;
$$;

GRANT EXECUTE ON FUNCTION public.get_due_cadence_agents() TO service_role;
```

Note: The original `get_due_heartbeat_agents()` is NOT dropped — it continues to work for the old heartbeat-runner path (use_langgraph=false users).
