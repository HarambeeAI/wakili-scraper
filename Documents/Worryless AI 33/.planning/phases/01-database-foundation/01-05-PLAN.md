<objective>
Security hardening of three existing edge functions (SEC-01) and creation of the shared workspace content sanitization utility (SEC-03). SEC-02's pattern is established by documentation — the actual dispatcher/runner functions are built in Phase 4.

Purpose: The existing `planning-agent`, `generate-leads`, and `crawl-business-website` functions accept `userId` from the request body — an attacker can spoof any user ID by crafting a request body. This fix extracts userId from the verified JWT in the Authorization header instead, making user identity tamper-proof. The sanitize module is created now so Phase 3 can import it without a dependency on Phase 3 planning.

Output: Three modified TypeScript files + one new shared utility module.

/**
 * Workspace Content Sanitization
 *
 * Strips known prompt injection patterns from user-authored workspace content.
 * Apply BEFORE storing to database AND before injecting into any LLM system prompt.
 *
 * SEC-03: Phase 1 — Database Foundation
 * Used by: Phase 3 workspace editor save, Phase 4 heartbeat runner
 */

export function sanitizeWorkspaceContent(content: string): string {
  const injectionPatterns: RegExp[] = [
    /ignore\s+(?:all\s+)?previous\s+instructions?/gi,
    /ignore\s+all\s+prior\s+instructions?/gi,
    /<\s*system\s*>/gi,
    /<\/\s*system\s*>/gi,
    /<\|im_start\|>/gi,
    /<\|im_end\|>/gi,
    /\[INST\]/gi,
    /\[\/INST\]/gi,
    /###\s*(?:instruction|system)/gi,
    /assistant:\s*you\s+are\s+now/gi,
    /you\s+are\s+now\s+(?:a\s+)?(?:different|new|another)\s+(?:ai|assistant|model)/gi,
    /disregard\s+(?:all\s+)?(?:previous|prior|earlier)\s+(?:instructions?|prompts?|context)/gi,
  ];

  let sanitized = content;
  for (const pattern of injectionPatterns) {
    sanitized = sanitized.replace(pattern, '[FILTERED]');
  }
  return sanitized.trim();
}
