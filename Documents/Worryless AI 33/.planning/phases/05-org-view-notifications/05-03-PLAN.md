    ```typescript
    export type HeartbeatStatus = 'active' | 'attention' | 'sleeping';

    export function getHeartbeatStatus(
      lastRunAt: string | null,
      lastOutcome: string | null
    ): HeartbeatStatus {
      if (!lastRunAt) return 'sleeping';
      if (lastOutcome === 'surfaced') return 'attention';
      const hoursSince = (Date.now() - new Date(lastRunAt).getTime()) / 3_600_000;
      if (hoursSince <= 1) return 'active';
      return 'sleeping';
    }
    ```
